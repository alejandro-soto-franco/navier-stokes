\chapter{Lean 4 Formalisation Reference}\label{app:lean}

This appendix documents the Lean~4 formalisation that accompanies the text.
Each section corresponds to a \texttt{.lean} source file in the repository;
the file links lead to the current version in the project repository.  All
definitions and theorem statements type-check against the pinned Mathlib4 fork
(commit \texttt{698d2b68}, based on \texttt{v4.29.0-rc8}) with 2753 build jobs
and zero errors.

Theorems are classified as \textbf{proved} (no \texttt{sorry}) or
\textit{sorry} stub (hypothesis stated precisely; proof deferred).  All
\textit{sorry} stubs correspond to Category~C obligations: standard PDE or
functional-analysis results not yet available in Mathlib that can in principle
be completed independently.

\subsection{Sobolev spaces}\label{sec:lean-ch1-sobolev}
% ---------------------------------------------------------------------------

\textbf{File:}
\href{https://github.com/alejandro-soto-franco/navier-stokes/blob/master/lean/NavierStokes/Foundations/SobolevSpace.lean}%
{\texttt{NavierStokes/Foundations/SobolevSpace.lean}}

\medskip\noindent
Defines the Sobolev spaces $W^{1,p}(\Omega)$, $H^{1}(\Omega)$, and
$H^{1}_{0}(\Omega)$; proves symmetry of the $H^{1}$ inner product.

\begin{itemize}
  \item \path{NavierStokes.SobolevW1p}: the Banach space $W^{1,p}(\Omega)$
    as a subtype of $L^{p}$ with $\norm{\cdot}_{W^{1,p}}$ norm
    (\Cref{def:W1p}).
  \item \path{NavierStokes.SobolevH1}: the Hilbert space $H^{1}(\Omega) =
    W^{1,2}(\Omega)$ with the standard inner product (\Cref{def:H1}).
  \item \path{NavierStokes.SobolevH1Zero}: the space $H^{1}_{0}(\Omega)$
    as the $H^{1}$-closure of $C^{\infty}_{c}(\Omega)$; the definition is
    sorry-free (\Cref{def:H10}).
  \item \path{NavierStokes.sobolevH1InnerProduct_comm} \textbf{(proved)}:
    symmetry of the $H^{1}$ inner product,
    $\langle u, v \rangle_{H^{1}} = \langle v, u \rangle_{H^{1}}$
    (\Cref{def:H1}).
\end{itemize}

\subsection{Sobolev embeddings}\label{sec:lean-ch1-embedding}
% ---------------------------------------------------------------------------

\textbf{File:}
\href{https://github.com/alejandro-soto-franco/navier-stokes/blob/master/lean/NavierStokes/Foundations/SobolevEmbedding.lean}%
{\texttt{NavierStokes/Foundations/SobolevEmbedding.lean}}

\medskip\noindent
States the Gagliardo--Nirenberg--Sobolev embedding and Morrey inequality, and
proves the conjugate exponent identities.

\begin{itemize}
  \item \path{NavierStokes.sobolevConjugate_gt} \textbf{(proved)}: the
    Sobolev conjugate satisfies $p^{*} > p$ for $1 \le p < n$
    (\Cref{thm:sobolev-subcritical}).
  \item \path{NavierStokes.sobolevConjugate_inv} \textbf{(proved)}: the
    dimensional identity $1/p^{*} = 1/p - 1/n$
    (\Cref{thm:sobolev-subcritical}).
  \item \path{NavierStokes.sobolev_embedding_subcritical}\\
    \emph{(\textit{sorry} stub)}: $W^{1,p}(\R^{n}) \hookrightarrow
    L^{p^{*}}(\R^{n})$ continuously for $1 \le p < n$
    (\Cref{thm:sobolev-subcritical}).
  \item \path{NavierStokes.sobolev_embedding_supercritical}\\
    \emph{(\textit{sorry} stub)}: $W^{1,p}(\R^{n}) \hookrightarrow
    C^{0,\alpha}(\R^{n})$ continuously for $p > n$, $\alpha = 1 - n/p$
    (\Cref{thm:sobolev-morrey}).
\end{itemize}

\subsection{Poincar\'{e} inequality}\label{sec:lean-ch1-poincare}
% ---------------------------------------------------------------------------

\textbf{File:}
\href{https://github.com/alejandro-soto-franco/navier-stokes/blob/master/lean/NavierStokes/Foundations/Poincare.lean}%
{\texttt{NavierStokes/Foundations/Poincare.lean}}

\medskip\noindent
States the Poincar\'{e} inequality and the Payne--Weinberger bound on the
optimal constant for convex domains.

\begin{itemize}
  \item \path{NavierStokes.poincare_inequality} \emph{(\textit{sorry} stub)}:
    for a bounded connected domain $\Omega \subset \R^{n}$, there exists
    $C_{P} > 0$ such that $\norm{u}_{L^{2}} \le C_{P}\norm{\nabla u}_{L^{2}}$
    for every $u \in H^{1}_{0}(\Omega)$.  The \texttt{IsConnected} hypothesis is
    required: without it, a constant-gradient argument fails on disconnected
    domains (\Cref{thm:poincare}).
  \item \path{NavierStokes.poincare_constant_bound_convex}
    \emph{(\textit{sorry} stub)}: for convex $\Omega$ the optimal constant
    satisfies $C_{P} \le \operatorname{diam}(\Omega)/\sqrt{n}$
    (Payne--Weinberger 1960) (\Cref{thm:poincare}).
\end{itemize}

\subsection{Divergence-free spaces and Helmholtz decomposition}%
\label{sec:lean-ch1-divfree}
% ---------------------------------------------------------------------------

\textbf{File:}
\href{https://github.com/alejandro-soto-franco/navier-stokes/blob/master/lean/NavierStokes/Foundations/DivFreeSpace.lean}%
{\texttt{NavierStokes/Foundations/DivFreeSpace.lean}}

\medskip\noindent
Defines the distributional divergence-free condition, the space $L^{2}_{\sigma}$,
and a legacy Leray projector stub; proves sequential closedness of $L^{2}_{\sigma}$
under $L^{2}$ convergence.

\begin{itemize}
  \item \path{NavierStokes.IsDistribDivFree}\\
    $u \in L^{2}$ is distributionally divergence-free, i.e.\
    $\int u \cdot \nabla\varphi\,\dd x = 0$ for all $\varphi \in C^{\infty}_{c}$
    (\Cref{def:distrib-divfree}).
  \item \path{NavierStokes.L2sigma}: the space $L^{2}_{\sigma}(\Omega)$ as
    the $L^{2}$-closure of smooth compactly supported divergence-free fields
    (\Cref{def:L2sigma}).
  \item \path{NavierStokes.l2sigma_closed_under_l2_convergence}\\
    \textbf{(proved)}: $L^{2}_{\sigma}$ is closed under $L^{2}$ convergence,
    via H\"{o}lder's inequality (Cauchy--Schwarz, $p = q = 2$) and uniqueness of
    limits (\Cref{prop:L2sigma-closed}).
  \item \path{NavierStokes.helmholtz_decomposition}
    \emph{(\textit{sorry} stub, legacy)}: every $u \in L^{2}(\Omega;\R^{n})$
    decomposes orthogonally as $u = v + \nabla p$ with $v \in L^{2}_{\sigma}$ and
    $\nabla p \perp L^{2}_{\sigma}$ (\Cref{thm:helmholtz}).  Retained for API
    compatibility; superseded by \path{helmholtz_l2_decomposition} in
    \Cref{sec:lean-ch1-helmholtz-proj}.
  \item \path{NavierStokes.lerayProjector_idempotent}\\
    \emph{(\textit{sorry} stub, legacy)}: $\mathbb{P}^{2} = \mathbb{P}$.
    Superseded by \path{lerayProjectorLp_idempotent} \textbf{(proved)}
    in \Cref{sec:lean-ch1-helmholtz-proj}.
  \item \path{NavierStokes.lerayProjector_selfAdjoint}\\
    \emph{(\textit{sorry} stub, legacy)}: $\mathbb{P}^{*} = \mathbb{P}$.
    Superseded by \path{lerayProjectorLp_selfAdjoint} \textbf{(proved)}
    in \Cref{sec:lean-ch1-helmholtz-proj}.
\end{itemize}

\subsection{Helmholtz projection via orthogonal projection}%
\label{sec:lean-ch1-helmholtz-proj}
% ---------------------------------------------------------------------------

\textbf{File:}
\href{https://github.com/alejandro-soto-franco/navier-stokes/blob/master/lean/NavierStokes/Foundations/HelmholtzProjection.lean}%
{\texttt{NavierStokes/Foundations/HelmholtzProjection.lean}}

\medskip\noindent
Constructs the Leray projector as the orthogonal projection onto the closed
submodule $L^{2}_{\sigma}$ using Mathlib's \path{Submodule.starProjection}
API.  All main theorems are sorry-free; one norm-bridge sorry remains
(Category~C, ENNReal bookkeeping).

\begin{itemize}
  \item \path{NavierStokes.l2sigmaSubmodule}\\
    $L^{2}_{\sigma}$ as a \path{Submodule} of $L^{p}(E,2,\mu)$; zero, add,
    and scalar-multiplication closures proved (\Cref{def:L2sigma}).
  \item \path{NavierStokes.l2sigmaSubmodule_isClosed}\\
    \textbf{(proved, modulo norm bridge)}: topological closedness via
    sequential closedness and the metric structure of $L^{2}$
    (\Cref{prop:L2sigma-closed}).
  \item \path{NavierStokes.l2sigmaClosedSubmodule}\\
    packages \path{l2sigmaSubmodule} as a \path{ClosedSubmodule},
    enabling Mathlib's \path{HasOrthogonalProjection} instance.
  \item \path{NavierStokes.lerayProjectorLp}\\
    the Leray projector $\mathbb{P}$ as a CLM onto $L^{2}_{\sigma}(\Omega)$,
    defined via \path{Submodule.starProjection} on \path{l2sigmaClosedSubmodule}
    (\Cref{def:leray-projector}).
  \item \path{NavierStokes.lerayProjectorLp_idempotent}\\
    \textbf{(proved)}: $\mathbb{P}(\mathbb{P}(f)) = \mathbb{P}(f)$, via
    \path{starProjection_eq_self_iff.mpr} and membership in $L^{2}_{\sigma}$
    (\Cref{prop:leray-props}).
  \item \path{NavierStokes.lerayProjectorLp_selfAdjoint}\\
    \textbf{(proved)}: $\langle \mathbb{P}f, g \rangle = \langle f, \mathbb{P}g \rangle$,
    via \path{inner_starProjection_left_eq_right} (\Cref{prop:leray-props}).
  \item \path{NavierStokes.helmholtz_l2_decomposition}\\
    \textbf{(proved)}: for every $f \in L^{2}(\Omega;\R^{n})$,
    $f = \mathbb{P}f + (f - \mathbb{P}f)$ with $\mathbb{P}f \in L^{2}_{\sigma}$
    and $\langle \mathbb{P}f, f - \mathbb{P}f \rangle = 0$, via
    \path{starProjection_inner_eq_zero} (\Cref{thm:helmholtz}).
\end{itemize}

\begin{itemize}
  \item \path{NavierStokes.trilinearForm}: the integral $b(u,v,w) = \int

    functions $\varphi_{k} = v_{k}^{2}$ (\Cref{lem:trilinear-antisymmetry}).
\end{itemize}

{\texttt{NavierStokes/LerayHopf/EnergyInequality.lean}}

\medskip\noindent
Defines the energy inequality satisfied by Leray--Hopf solutions.

\begin{itemize}
  \item \path{NavierStokes.SatisfiesEnergyInequality}: the weak solution $u$
    satisfies
    \[
      \norm{u(t)}_{L^{2}}^{2}
      + 2\nu\int_{0}^{t}\norm{\nabla u(s)}_{L^{2}}^{2}\,\dd s
      \;\le\; \norm{u_{0}}_{L^{2}}^{2}
    \]
    for almost every $t \ge 0$ (\Cref{thm:energy-inequality}).
\end{itemize}
